function spinWords(string){
  string = string.split(' ');
  let result = [];
  for(let i = 0; i < string.length; i++) {
    string[i].length >= 5 ? result.push(string[i].split('').reverse().join('')) : result.push(string[i])
  }
  return result.join(' ');
}
